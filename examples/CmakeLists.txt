cmake_minimum_required(VERSION 3.5...4.3.1)

project(stochastic_simlib_examples LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT TARGET stochastic)
    add_subdirectory(../lib ${CMAKE_BINARY_DIR}/stochastic_simlib_build)
endif()

add_library(models models.cpp)

add_executable(observer
    observer.cpp
)

add_executable(visitor
    visitor.cpp
)

add_executable(plot
    plot.cpp
)

add_executable(multi
    multi.cpp
)

target_link_libraries(models PRIVATE stochastic)
target_link_libraries(observer PRIVATE stochastic models)
target_link_libraries(visitor PRIVATE stochastic models)
target_link_libraries(plot PRIVATE stochastic models)
target_link_libraries(multi PRIVATE stochastic models)
